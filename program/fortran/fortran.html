Quadratic Equation

Here we have a simple code for solving a quadratic equation using only real variables in the code. As we know, the genalizated solution for quadratic equations are described by: $$x = {-b \pm \sqrt{b^2-4ac} \over 2a}.$$ Therefore, is so easy compute the value for any QE. 
PROGRAM QuadraticEquation
IMPLICIT NONE

REAL*8:: a,b,c                          !Real, for ax^2 + bx + c = 0
REAL*8:: d                              !Discriminant
REAL*8:: root1,root2,root3,root4        !Roots, for i we'll use the character 'i'

WRITE (UNIT=*, FMT='(/A/)') 'Write down coeficients as: ax^2 + bx + c = 0:'

WRITE (UNIT=*, FMT='(a)', ADVANCE='NO') ' a = '
READ(*,*) a

WRITE (UNIT=*, FMT='(a)', ADVANCE='NO') ' b = '
READ(*,*) b

WRITE (UNIT=*, FMT='(a)', ADVANCE='NO') ' c = '
READ(*,*) c

!Can we compute?

IF (a == 0.0) THEN
    IF (b == 0.0) THEN
        IF (c == 0.0) THEN
            WRITE (*,*) 'Come on, dude. All the numbers are roots! LOL' !0=0 tautology
        ELSE
            WRITE (*,*) 'Unsolve Equation, bist du blau? LMAO'          !Rlly!?
        END IF
    ELSE
        WRITE(*,*) 'Is a Linear Equation, but the root is: ', -c/b      !not coefficient for x^2
    END IF
ELSE 

!Here we go with the chicharronera

d = b*b - 4.D0*a*c                !Discriminant

IF ( d >= 0.0 ) THEN               !We'll have real roots
    d     = SQRT(d)
    root1 = (-b + d)/(2.D0*a)     ! first root
    root2 = (-b - d)/(2.D0*a)     ! second root
    WRITE(*,FMT="(A,F9.5,A,F9.5)")  'Roots are: ', root1, ' and ', root2
ELSE                              !We'll have complex roots
    d     = SQRT(-d)
    root3 = -b/(2.D0*a)           !roots as parts of whole root
    root4 =  d/(2.D0*a)
    WRITE(*,*)  'The roots are:'
    WRITE(*,FMT="(F9.5,A3,F9.5,A)")  root3,'  +', root4,'i'
    WRITE(*,FMT="(F9.5,A3,F9.5,A)")  root3,'  -', root4,'i'

END IF
END IF

END PROGRAM QuadraticEquation
